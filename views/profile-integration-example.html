Integration Example

// 1. Include the component script
<script src="js/profile-settings-component.js"></script>

// 2. Add a button to trigger the modal
<button onclick="window.profileSettingsModal.open()">
    Profile Settings
</button>

// 3. Customize with options (optional)
const profileModal = new ProfileSettingsModal({
    accentColor: '#4361EE',
    apiEndpoint: '/api/profile'
});

// 4. Listen for profile updates
window.addEventListener('profileUpdated', (event) => {
    console.log('Profile updated:', event.detail);
});
